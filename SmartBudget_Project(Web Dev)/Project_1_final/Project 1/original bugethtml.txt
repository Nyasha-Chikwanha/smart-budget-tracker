  <script>
    const form = document.getElementById('budgetForm');
    const transactionsTable = document.getElementById('transactionsTable').querySelector('tbody');
    const totalIncome = document.getElementById('totalIncome');
    const totalExpenses = document.getElementById('totalExpenses');
    const balance = document.getElementById('balance');
    const progressBar = document.getElementById('progressBar');

    let income = 0;
    let expenses = 0;

    form.addEventListener('submit', (e) => {
      e.preventDefault();
      const type = document.getElementById('type').value;
      const category = document.getElementById('category').value;
      const amount = parseFloat(document.getElementById('amount').value);

      const row = document.createElement('tr');
      row.innerHTML = `
        <td>${type}</td>
        <td>${category}</td>
        <td>$${amount.toFixed(2)}</td>
        <td><button onclick="removeRow(this)">Delete</button></td>
      `;
      transactionsTable.appendChild(row);

      if (type === 'income') income += amount;
      else expenses += amount;

      updateSummary();
      form.reset();
    });

    function removeRow(btn) {
      const row = btn.parentElement.parentElement;
      const type = row.children[0].textContent;
      const amount = parseFloat(row.children[2].textContent.replace('$', ''));

      if (type === 'income') income -= amount;
      else expenses -= amount;

      row.remove();
      updateSummary();
    }

    function updateSummary() {
      const currentBalance = income - expenses;
      const usedBudget = (expenses / income) * 100;

      totalIncome.textContent = `$${income.toFixed(2)}`;
      totalExpenses.textContent = `$${expenses.toFixed(2)}`;
      balance.textContent = `$${currentBalance.toFixed(2)}`;

      progressBar.style.width = `${usedBudget.toFixed(0)}%`;
      progressBar.textContent = `${usedBudget.toFixed(0)}%`;
    }
  </script>
